  • modal
    button[data-toggle="modal"][data-target="#xxxxx"] : 启动modal
    a[data-toggle="modal"][href="#xxxxx"] : 启动modal
    .modal.fade[role="dialog"]>.modal-dialog[role="document"]>.modal-content>(.modal-header+.modal-body+.modal-footer)
    .modal-dialog(.modal-lg、.modal-sm)
    .modal-header.modal-title : 标题
    .modal[data-dismiss="modal"] : 关闭modal
    .modal-header>.close[data-dismiss="modal"]{&times;}
  • modal方法
    $('#modal1').modal('toggle') : 手动打开或关闭modal，触发 "shown.bs.modal" 或 "hidden.bs.modal"
    $('#modal1').modal('show') : 显示modal，触发 "shown.bs.modal"
    $('#modal1').modal('hide') : 隐藏modal，触发 "hidden.bs.modal"
    $('#modal1').modal({
    "keyboard" : "true", // 键盘上的 esc 键被按下时关闭模态框
    "show" : "true" // 模态框初始化之后就立即显示出来
});
  • modal 事件
    $('#modal1').on('event', function(event){var ref=$(event.relatedTarget);});
    "show.bs.modal" : show 方法调用之后立即触发该事件，如果是通过点击某个作为触发器的元素，则此元素可以通过事件的 relatedTarget 属性进行访问
    "shown.bs.modal" : 此事件在模态框已经显示出来（并且同时在 CSS 过渡效果完成）之后被触发。如果是通过点击某个作为触发器的元素，则此元素可以通过事件的 relatedTarget 属性进行访问。
    "hide.bs.modal" : hide 方法调用之后立即触发该事件
    "hidden.bs.modal" : 此事件在模态框被隐藏（并且同时在 CSS 过渡效果完成）之后被触发
    "loaded.bs.modal" : 从远端的数据源加载完数据之后触发该事件
