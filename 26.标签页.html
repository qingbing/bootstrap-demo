  • 标签页事件
    $('a[data-toggle="tab"]').on('shown.bs.tab', function (e) {
  e.target // newly activated tab
  e.relatedTarget // previous active tab
})
    "show.bs.tab" : 当标签 展开 时触发的事件
    "shown.bs.tab" : 当标签 展开后 时触发的事件
    "hide.bs.tab" : 当标签 隐藏 时触发的事件
    "hidden.bs.tab" : 当标签 隐藏后 时触发的事件
